## 풀이

```python

def reverse(x):
    res = 0
    while x > 0:
        t = x % 10
        res = res * 10 + t  # 외우자
        x = x // 10
    return res


def isPrime(x):
    if x == 1:
        return False
    for i in range(2, x // 2 + 1):
        if x % i == 0:
            return False
    else:
        return True


def isPrime(*args):

    for x in args:
        tmp = reverse(x)
        if isPrime(tmp):
            print(tmp, end=" ")


isPrime(32, 55, 62, 3700, 250)
```
